ls - показать все файлы и папки
cd - переход в директорию
pwd - показывает 
git config --global user.name/email "..." - добавление имени и почты
git init - инициализирует репозиторий в текущей пустой папке
ls -a - "-а" показывает скрытые папки
git status - показывает статус файлов в репозитории
git add - передаёт файл под управление git
git commit -m "..." - добавляет комментарий
git log - лог комментариев
git push - передача файлов в удалённый репозиторий
